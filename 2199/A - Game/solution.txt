fun main() {    val br = System.`in`.bufferedReader()    val t = br.readLine().trim().toInt()     repeat(t) {        val k = br.readLine().trim().toInt()         val (a1, b1) = br.readLine().trim().split(" ").map { it.toInt() }        val (a2, b2) = br.readLine().trim().split(" ").map { it.toInt() }         val aliWins = (if (a1 > b1) 1 else 0) + (if (a2 > b2) 1 else 0)        val bobWins = (if (b1 > a1) 1 else 0) + (if (b2 > a2) 1 else 0)         if (a1 + a2 < b1 + b2 + k) {            println("YES")        } else if (a1 + a2 == b1 + b2 + k) {            println(if (aliWins <= bobWins) "YES" else "NO")        } else {            println("NO")        }    }}
